public class ThreadMethodsExample {
public static void main(String[] args) {
    Thread thread = new Thread(() -> {
        for (int i = 1; i <= 5; i++) {
            System.out.println("Thread: " + Thread.currentThread().getName() + ", Count: " + i);
            try {
                Thread.sleep(1000); // Introduce a delay
            } catch (InterruptedException e) {
                System.out.println("Thread interrupted");
            }
        }
    });

    thread.setName("MyThread"); // Set thread name
    thread.setPriority(Thread.MAX_PRIORITY); // Set thread priority

    thread.start(); // Start the thread

    // Wait for the thread to complete its execution
    try {
        thread.join();
    } catch (InterruptedException e) {
        System.out.println("Main thread interrupted");
    }

    // Check if the thread is alive
    if (thread.isAlive()) {
        System.out.println("Thread is still alive");
    } else {
        System.out.println("Thread has completed its execution");
    }
}
}
